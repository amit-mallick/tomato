def chennai
  # render text: "chennai"
  @resturants = Resturant.all
  @location = Location.where(location: "Chennai").all
  # @location = Location.find(4)
  @filter = []
  for r in @resturants
    for l in @location
      if r.id == l.resturant_id
        @filter << r
      end
    end
  end
end
